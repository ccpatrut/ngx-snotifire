import {
  ChangeDetectionStrategy,
  Component,
  Input,
  OnInit,
  ViewEncapsulation,
} from '@angular/core';
import { SnotifireToastModel } from '../toast/notifire-toast.model';
import { SnotifireService } from '../../services';

@Component({
  selector: 'notifire-button',
  templateUrl: './buttons.component.html',
  changeDetection: ChangeDetectionStrategy.OnPush,
  encapsulation: ViewEncapsulation.None,
})
/**
 * Buttons component
 */
export class ButtonsComponent {
  /**
   * Get buttons Array
   */
  @Input() toast!: SnotifireToastModel;

  constructor(private readonly service: SnotifireService) {}

  /**
   * remove toast
   */
  remove() {
    this.service.remove(this.toast.id);
  }
}

<div class="notifire-toast__buttons">
    <ng-container *ngIf="toast.config">
        <button
            type="button"
            *ngFor="let button of toast.config.buttons"
            [ngClass]="{ 'notifire-toast__buttons--bold': button.bold }"
            (click)="button.action ? button.action(toast) : remove()"
        >
            {{ button.text }}
        </button>
    </ng-container>
</div>
