  • success:"success"
  • error:"error"
  • warning:"warning"
  • info:"info"
  • async:"async"
  • confirm:"confirm"
  • prompt:"prompt"
Example
import { SnotificationPositionType, SnotifireType } from "ng-snotify";

snotifireService.create({
  title: "Example title",
  body: null,
  config: {
    position: SnotifyPosition.rightTop,
    type: SnotifyStyle.info,
  },
});
